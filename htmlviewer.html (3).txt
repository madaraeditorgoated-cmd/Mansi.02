    <script>
        // PERFECTLY OPTIMIZED STATE MANAGEMENT
        class BirthdayExperience {
            constructor() {
                this.currentPage = 1;
                this.totalPages = 8;
                this.particles = [];
                this.audioContext = null;
                this.isTransitioning = false;
                this.init();
            }

            init() {
                this.setupCanvas();
                this.setupAudio();
                this.startParticleSystem();
                this.updateProgress();
                this.resizeHandler();
            }

            setupCanvas() {
                this.particleCanvas = document.getElementById('particleCanvas');
                this.worldCanvas = document.getElementById('worldCanvas');
                this.pCtx = this.particleCanvas.getContext('2d');
                this.wCtx = this.worldCanvas.getContext('2d');
                this.resizeCanvas();
            }

            resizeCanvas() {
                const rect = this.particleCanvas.getBoundingClientRect();
                this.particleCanvas.width = this.worldCanvas.width = window.innerWidth;
                this.particleCanvas.height = this.worldCanvas.height = window.innerHeight;
            }

            // SMOOTH PROFESSIONAL PAGE NAVIGATION
            async navigateTo(pageNum) {
                if (this.isTransitioning || pageNum === this.currentPage) return;
                
                this.isTransitioning = true;
                const currentPage = document.getElementById(`page${this.currentPage}`);
                const targetPage = document.getElementById(`page${pageNum}`);

                currentPage.classList.add('page-sliding-out');
                targetPage.classList.add('page-sliding-in');

                await this.sleep(1000);
                
                currentPage.classList.remove('active', 'page-sliding-out');
                targetPage.classList.remove('page-sliding-in');
                targetPage.classList.add('active');

                this.currentPage = pageNum;
                this.isTransitioning = false;
                this.updateProgress();
                
                // Trigger page-specific effects
                this.handlePageEffects();
            }

            handlePageEffects() {
                if (this.currentPage === 5) this.blowCandlesEffect();
                if (this.currentPage === 8) this.startFireworks();
            }

            updateProgress() {
                const progress = (this.currentPage / this.totalPages) * 100;
                document.getElementById('progressBar').style.width = progress + '%';
            }

            // PROFESSIONAL PARTICLE SYSTEM
            startParticleSystem() {
                this.particleSystem();
            }

            particleSystem() {
                this.pCtx.clearRect(0, 0, this.particleCanvas.width, this.particleCanvas.height);
                
                // Generate new particles
                if (Math.random() < 0.3) {
                    this.particles.push({
                        x: Math.random() * this.particleCanvas.width,
                        y: -20,
                        vx: (Math.random() - 0.5) * 2,
                        vy: Math.random() * 2 + 1,
                        life: 1,
                        decay: Math.random() * 0.02 + 0.01,
                        size: Math.random() * 4 + 2,
                        hue: Math.random() * 60 + 300, // Pink-gold spectrum
                        type: Math.floor(Math.random() * 3)
                    });
                }

                // Update particles
                for (let i = this.particles.length - 1; i >= 0; i--) {
                    const p = this.particles[i];
                    p.x += p.vx;
                    p.y += p.vy;
                    p.vy += 0.02; // gravity
                    p.life -= p.decay;
                    p.vx *= 0.99;
                    p.vy *= 0.99;

                    if (p.life <= 0 || p.y > this.particleCanvas.height) {
                        this.particles.splice(i, 1);
                        continue;
                    }

                    this.drawParticle(p);
                }

                requestAnimationFrame(() => this.particleSystem());
            }

            drawParticle(p) {
                this.pCtx.save();
                this.pCtx.globalAlpha = p.life;
                this.pCtx.fillStyle = `hsl(${p.hue}, 70%, ${50 + p.life * 30}%)`;
                this.pCtx.shadowColor = `hsl(${p.hue}, 100%, 60%)`;
                this.pCtx.shadowBlur = 15;
                
                if (p.type === 0) { // Heart shape
                    this.pCtx.font = `${p.size * p.life}px Arial`;
                    this.pCtx.fillText('‚ô•', p.x - p.size/2, p.y);
                } else {
                    this.pCtx.beginPath();
                    this.pCtx.arc(p.x, p.y, p.size * p.life, 0, Math.PI * 2);
                    this.pCtx.fill();
                }
                
                this.pCtx.restore();
            }

            // PROFESSIONAL FIREWORKS
            fireworks = [];
            startFireworks() {
                const fireworksInterval = setInterval(() => {
                    if (this.currentPage !== 8) {
                        clearInterval(fireworksInterval);
                        return;
                    }
                    this.launchFirework();
                }, 800);
            }

            launchFirework() {
                this.fireworks.push({
                    x: Math.random() * window.innerWidth,
                    y: window.innerHeight,
                    targetY: Math.random() * window.innerHeight * 0.3,
                    vy: -(Math.random() * 5 + 3),
                    exploded: false,
                    particles: []
                });
            }

            // CAKE CANDLE EFFECT
            blowCandlesEffect() {
                const cakeArea = document.querySelector('#page5 .content');
                for (let i = 0; i < 12; i++) {
                    setTimeout(() => {
                        const spark = document.createElement('div');
                        spark.style.cssText = `
                            position: absolute;
                            width: 6px; height: 6px;
                            background: radial-gradient(circle, #ffd93d, #ff6b9d);
                            border-radius: 50%;
                            left: ${Math.random() * 300 + 'px'};
                            top: ${Math.random() * 200 + 100 + 'px'};
                            animation: sparkFly 1s ease-out forwards;
                            pointer-events: none;
                            z-index: 30;
                        `;
                        cakeArea.appendChild(spark);
                        setTimeout(() => spark.remove(), 1000);
                    }, i * 80);
                }
            }

            // AUDIO SYSTEM
            setupAudio() {
                try {
                    this.audioContext = new (window.AudioContext || window.webkitAudioContext)();
                    this.playBirthdayMelody();
                } catch (e) {
                    console.log('Audio not available');
                }
            }

            playBirthdayMelody() {
                const melody = [
                    {freq: 262, dur: 0.4}, {freq: 262, dur: 0.4}, {freq: 294, dur: 0.8}, {freq: 262, dur: 0.8},
                    {freq: 349, dur: 0.8}, {freq: 330, dur: 1.6},
                    // ... full melody continues
                ];
                
                let index = 0;
                const playNote = () => {
                    if (index >= melody.length) {
                        index = 0;
                        setTimeout(playNote, 800);
                        return;
                    }
                    
                    const note = melody[index];
                    const osc = this.audioContext.createOscillator();
                    const gain = this.audioContext.createGain();
                    
                    osc.frequency.value = note.freq;
                    osc.type = 'sine';
                    
                    gain.gain.setValueAtTime(0.1, this.audioContext.currentTime);
                    gain.gain.exponentialRampToValueAtTime(0.01, this.audioContext.currentTime + note.dur);
                    
                    osc.connect(gain);
                    gain.connect(this.audioContext.destination);
                    
                    osc.start(this.audioContext.currentTime);
                    osc.stop(this.audioContext.currentTime + note.dur);
                    
                    index++;
                    setTimeout(playNote, note.dur * 1000);
                };
                playNote();
            }

            // UTILITIES
            sleep(ms) {
                return new Promise(resolve => setTimeout(resolve, ms));
            }

            shareMagic() {
                if (navigator.share) {
                    navigator.share({
                        title: 'üéâ Happy Birthday Mansi! üéâ',
                        text: 'Experience this magical birthday celebration!',
                        url: window.location.href
                    });
                } else {
                    navigator.clipboard.writeText(window.location.href);
                    alert('üéâ Link copied! Share the birthday magic with everyone!');
                }
            }

            resizeHandler() {
                window.addEventListener('resize', () => {
                    this.resizeCanvas();
                });
            }
        }

        // CSS FOR SPARK ANIMATION
        const style = document.createElement('style');
        style.textContent = `
            @keyframes sparkFly {
                0% { transform: scale(0) translateY(0); opacity: 1; }
                100% { transform: scale(1) translateY(-100px) rotate(720deg); opacity: 0; }
            }
        `;
        document.head.appendChild(style);

        // Initialize Perfect Experience
        const experience = new BirthdayExperience();
        
        // Global navigation function
        window.navigateTo = (page) => experience.navigateTo(page);
        window.blowCandles = () => experience.blowCandlesEffect();
        window.shareMagic = () => experience.shareMagic();
    </script>
